## Imports

```tsx
import { PrimeCard } from '@components/shared';
```

## Code

```tsx
// Configurable usage (shorthand props)
<PrimeCard title="Card Title" icon={<Icon />} tooltip="Hint text" description="Short ref" variant="primary">
  Content
</PrimeCard>

// Compound usage (full control)
<PrimeCard variant="primary">
  <PrimeCard.Header>
    <PrimeCard.Leading>
      <Icon />

      <PrimeCard.Title>
        Title
        <PrimeCard.Tooltip>Hint text for longer explanations</PrimeCard.Tooltip>
      </PrimeCard.Title>

      <PrimeCard.Description>Short reference (few words)</PrimeCard.Description>
    </PrimeCard.Leading>
  </PrimeCard.Header>

  <PrimeCard.Content>Content</PrimeCard.Content>
</PrimeCard>

// Available variants
variant="default" | "primary" | "secondary" | "success" | "destructive" | "warning" | "info" |
        "outline" | "outline-secondary" | "glass" | "gradient"
```
