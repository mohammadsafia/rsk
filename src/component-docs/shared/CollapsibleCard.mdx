import { CollapsibleCard } from '@components/shared';

## Code

```tsx
// Basic usage with clickable title and default TriggerIcon
<CollapsibleCard>
  <CollapsibleCard.Header>
    <CollapsibleCard.Trigger className="flex w-full items-center justify-between">
      <CollapsibleCard.Title>Title</CollapsibleCard.Title>
      <CollapsibleCard.TriggerIcon />
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>
    Content here
  </CollapsibleCard.Content>
</CollapsibleCard>

// With Leading + Tooltip (icon, title, and hover hint)
<CollapsibleCard variant="primary">
  <CollapsibleCard.Header>
    <CollapsibleCard.Leading>
      <Icon />

      <CollapsibleCard.Title>
        Title
        <CollapsibleCard.Tooltip>Hint text for longer explanations</CollapsibleCard.Tooltip>
      </CollapsibleCard.Title>
    </CollapsibleCard.Leading>

    <CollapsibleCard.Trigger>
      <CollapsibleCard.TriggerIcon />
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// With Leading + Description (icon, title, and short always-visible reference)
<CollapsibleCard variant="primary">
  <CollapsibleCard.Header>
    <CollapsibleCard.Leading>
      <Icon />

      <CollapsibleCard.Title>Title</CollapsibleCard.Title>

      <CollapsibleCard.Description>Short reference (few words)</CollapsibleCard.Description>
    </CollapsibleCard.Leading>

    <CollapsibleCard.Trigger>
      <CollapsibleCard.TriggerIcon />
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// Override TriggerIcon with custom icon and clickable title
<CollapsibleCard variant="primary">
  <CollapsibleCard.Header>
    <CollapsibleCard.Trigger className="flex w-full items-center justify-between">
      <CollapsibleCard.Title>Title</CollapsibleCard.Title>
      <CollapsibleCard.TriggerIcon>
        <Plus className="h-4 w-4 transition-transform duration-200 group-data-[state=open]:rotate-45" />
      </CollapsibleCard.TriggerIcon>
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// Controlled state
<CollapsibleCard open={isOpen} onOpenChange={setIsOpen}>
  ...
</CollapsibleCard>

// Available variants
variant="default" | "primary" | "secondary" | "success" | "destructive" | "warning" | "info" |
        "outline" | "outline-secondary" | "glass" | "gradient"
```
