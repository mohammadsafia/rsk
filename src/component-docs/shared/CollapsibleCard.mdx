## Imports

```tsx
import { CollapsibleCard } from '@components/shared';
```

## Code

```tsx
// Basic usage with clickable title and default TriggerIcon
<CollapsibleCard>
  <CollapsibleCard.Header>
    <CollapsibleCard.Trigger className="flex w-full items-center justify-between">
      <CollapsibleCard.Title>Title</CollapsibleCard.Title>
      <CollapsibleCard.TriggerIcon />
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>
    Content here
  </CollapsibleCard.Content>
</CollapsibleCard>

// Override TriggerIcon with custom icon and clickable title
<CollapsibleCard variant="primary">
  <CollapsibleCard.Header>
    <CollapsibleCard.Trigger className="flex w-full items-center justify-between">
      <CollapsibleCard.Title>Title</CollapsibleCard.Title>
      <CollapsibleCard.TriggerIcon>
        <Plus className="h-4 w-4 transition-transform duration-200 group-data-[state=open]:rotate-45" />
      </CollapsibleCard.TriggerIcon>
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// Another custom icon example
<CollapsibleCard variant="warning">
  <CollapsibleCard.Header>
    <CollapsibleCard.Title>Title</CollapsibleCard.Title>
    <CollapsibleCard.Trigger>
      <CollapsibleCard.TriggerIcon>
        <Plus className="h-4 w-4 transition-transform duration-200 group-data-[state=open]:rotate-90" />
      </CollapsibleCard.TriggerIcon>
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// TriggerIcon with text or custom elements
<CollapsibleCard variant="success">
  <CollapsibleCard.Header>
    <CollapsibleCard.Title>Title</CollapsibleCard.Title>
    <CollapsibleCard.Trigger>
      <CollapsibleCard.TriggerIcon>
        <span>Show/Hide</span>
      </CollapsibleCard.TriggerIcon>
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// Everything in the trigger - icon pushed to the end
<CollapsibleCard variant="info">
  <CollapsibleCard.Header>
    <CollapsibleCard.Trigger className="flex w-full items-center gap-2">
      <Icon className="h-5 w-5" />
      <CollapsibleCard.Title>Clickable Title</CollapsibleCard.Title>
      <CollapsibleCard.TriggerIcon className="ml-auto" />
    </CollapsibleCard.Trigger>
  </CollapsibleCard.Header>
  <CollapsibleCard.Content>Content</CollapsibleCard.Content>
</CollapsibleCard>

// Controlled state
<CollapsibleCard open={isOpen} onOpenChange={setIsOpen}>
  ...
</CollapsibleCard>

// Available variants
variant="default" | "primary" | "secondary" | "success" | "danger" | "warning" | "info" |
        "outline" | "outline-secondary" | "flat" | "elevated" | "elevated-lg" |
        "elevated-xl" | "glass" | "gradient"
```
