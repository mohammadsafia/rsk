## Code

```tsx

<FileUploader name="document" value={file} onChange={setFile}>
  <FileUploader.DropzoneGroup>
    <FileUploader.Dropzone accept=".pdf,.doc" placeholder="Drop file here" />
    <FileUploader.Actions />
  </FileUploader.DropzoneGroup>
</FileUploader>

// With file list and image preview
<FileUploader name="files" value={files} onChange={setFiles} multiple>
  <FileUploader.DropzoneGroup>
    <FileUploader.Dropzone accept="image/*" />

</FileUploader>
```
