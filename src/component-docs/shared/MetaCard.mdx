## Imports

```tsx
import { MetaCard } from '@components/shared';
```

## Code

```tsx
{
  /* Default vertical */
}
<MetaCard title="Title" subtitle="Subtitle" description="Description" />;

{
  /* With truncation */
}
<MetaCard title="Email" truncate description="very.long.email@example.com" />;

{
  /* Horizontal direction */
}
<MetaCard direction="horizontal" title="Label" description="Value" />;

{
  /* With image */
}
<MetaCard title="Avatar" imageURL="/path/to/image.png" />;

{
  /* Loading */
}
<MetaCard isLoading />;
```
