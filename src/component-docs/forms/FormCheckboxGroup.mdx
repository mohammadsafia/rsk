## Imports

```tsx
import { FormContainer, FormCheckboxGroup } from '@components/forms';
import { useForm } from 'react-hook-form';
```

## Code

```tsx
const form = useForm({ defaultValues: { features: ['dark-mode'] } });
const features = [
  { id: 'dark-mode', name: 'Dark Mode' },
  { id: 'notifications', name: 'Notifications' },
  { id: 'analytics', name: 'Analytics' },
];

<FormContainer formContext={form} onSuccess={console.log}>
  <FormCheckboxGroup
    name="features"
    control={form.control}
    label="Select Features"
    required
    options={features}
    getOptionLabel={(o) => o.name}
    getOptionValue={(o) => o.id}
  />
  <Button type="submit">Submit</Button>
</FormContainer>;
```
