## Imports

```tsx
import { FormContainer, FormMultiCombobox } from '@components/forms';
import { useForm } from 'react-hook-form';
```

## Code

```tsx
const form = useForm({ defaultValues: { skills: [] } });
const skills = [
  { id: 'js', name: 'JavaScript' },
  { id: 'ts', name: 'TypeScript' },
];

<FormContainer formContext={form} onSuccess={console.log}>
  <FormMultiCombobox
    name="skills"
    control={form.control}
    label="Skills"
    options={skills}
    getOptionLabel={(o) => o.name}
    getOptionValue={(o) => o.id}
    placeholder="Select skills..."
  />
</FormContainer>;
```
