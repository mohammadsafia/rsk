## Imports

```tsx
import { FormContainer, FormCombobox } from '@components/forms';
import { useForm } from 'react-hook-form';
import { LookupService } from 'api';
```

## Code

```tsx
const form = useForm();

<FormContainer formContext={form} onSuccess={console.log}>
  <FormCombobox
    name="userId"
    control={form.control}
    label="User"
    options={LookupService.users.request}
    getOptionLabel={(o) => o.name || o.label || String(o)}
    getOptionValue={(o) => o.id || String(o)}
    placeholder="Search users..."
    urlSearchParams={new URLSearchParams({ roles: 'admin,user' })}
    debounceMs={500}
    pageSize={20}
  />
</FormContainer>;
```
