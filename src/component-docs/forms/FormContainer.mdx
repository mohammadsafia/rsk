## Imports

```tsx
import { FormContainer } from '@components/forms';
import { useForm } from 'react-hook-form';
```

## Code

```tsx
const form = useForm({ defaultValues: { name: '' } });

<FormContainer formContext={form} onSuccess={(data) => console.log(data)}>
  {/* Form fields go here */}
</FormContainer>;
```
