## Imports

```tsx
import { FormContainer, FormRadioGroup } from '@components/forms';
import { useForm } from 'react-hook-form';
```

## Code

```tsx
const form = useForm({ defaultValues: { plan: 'basic' } });
const plans = [
  { id: 'basic', name: 'Basic Plan' },
  { id: 'pro', name: 'Pro Plan' },
  { id: 'enterprise', name: 'Enterprise Plan' },
];

<FormContainer formContext={form} onSuccess={console.log}>
  <FormRadioGroup
    name="plan"
    control={form.control}
    label="Select Plan"
    required
    options={plans}
    getOptionLabel={(o) => o.name}
    getOptionValue={(o) => o.id}
  />
  <Button type="submit">Submit</Button>
</FormContainer>
```
