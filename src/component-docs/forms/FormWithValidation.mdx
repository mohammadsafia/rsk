## Imports

```tsx
import { FormContainer, FormInput, FormNumber, Button } from '@components/forms';
import { useForm } from 'react-hook-form';
import { z } from 'zod';
```

## Code

```tsx
const schema = z.object({
  email: z.string().email('Invalid email'),
  password: z.string().min(8, 'Min 8 chars'),
  age: z.number().min(18, 'Must be 18+'),
});

type FormData = z.infer<typeof schema>;

const form = useForm<FormData>({
  defaultValues: { email: '', password: '', age: 18 },
});

<FormContainer formContext={form} onSuccess={console.log}>
  <FormInput name="email" label="Email" required />
  <FormInput name="password" type="password" label="Password" required />
  <FormNumber name="age" control={form.control} label="Age" />
  <Button type="submit">Submit</Button>
</FormContainer>;
```
