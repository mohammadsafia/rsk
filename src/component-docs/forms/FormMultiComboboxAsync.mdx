## Imports

```tsx
import { FormContainer, FormMultiCombobox } from '@components/forms';
import { useForm } from 'react-hook-form';
```

## Code

```tsx
const form = useForm({ defaultValues: { userIds: [] } });

<FormContainer formContext={form} onSuccess={console.log}>
  <FormMultiCombobox
    name="userIds"
    control={form.control}
    label="Users"
    options={LookupService.users.request}
    getOptionLabel={(o) => o.name || o.label || String(o)}
    getOptionValue={(o) => o.id || String(o)}
    placeholder="Search and select users..."
    urlSearchParams={new URLSearchParams({ roles: 'admin,user' })}
    debounceMs={500}
    pageSize={20}
    valueType="flat"
  />
</FormContainer>;
```
